public static FirstDocument AsDocument(this First entity)
    => new FirstDocument
    {
      Id = entity.Id,
Name = entity.Name,
Value = entity.Value
    };
    public static FirstDto AsDto(this FirstDocument document)
        => new FirstDto
        {
          Id = document.Id,
Name = document.Name,
Value = document.Value
        };
    public static First AsEntity(this FirstDocument document)
      => new First(document.Id,document.Name,document.Value);
    public static First AsEntity(this FirstDto dto)
      => new First(dto.Id,dto.Name,dto.Value);
